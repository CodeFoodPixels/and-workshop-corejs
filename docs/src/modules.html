    A basic module

  • §

    Is just an object with properties applied.

    const databaseSeed = {
  init: function(query) {
    const patch = this.getPatch();
    return this.runQuery(patch);
  },

  getPatch() {
    return 'INSERT into USERS ...';
  },

  runQuery(query) {
    return `Running: ${query}`;
  }
};

console.log(databaseSeed.init());

    return `Running: ${query}`;
  };

  const init = function() {
    const patch = getPatch();
